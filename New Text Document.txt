script type=module
     --- THREE.JS BACKGROUND HEARTS ---
    let scene, camera, renderer, hearts = [];
    let finaleTriggered = false;

    function initThree() {
        const canvas = document.getElementById('bg-canvas');
        scene = new THREE.Scene();
        camera = new THREE.PerspectiveCamera(75, window.innerWidth  window.innerHeight, 0.1, 1000);
        camera.position.z = 10;

        renderer = new THREE.WebGLRenderer({ canvas, alpha true });
        renderer.setSize(window.innerWidth, window.innerHeight);

        const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
        scene.add(ambientLight);

        const directionalLight = new THREE.DirectionalLight(0xffc0cb, 1);
        directionalLight.position.set(5, 5, 5);
        scene.add(directionalLight);

        createHearts();
        animate();
    }

    function createHearts() {
        const heartShape = new THREE.Shape();
        const x = -2.5, y = -5;
        heartShape.moveTo(x + 2.5, y + 2.5);
        heartShape.bezierCurveTo(x + 2.5, y + 2.5, x + 2, y, x, y);
        heartShape.bezierCurveTo(x - 3, y, x - 3, y + 3.5, x - 3, y + 3.5);
        heartShape.bezierCurveTo(x - 3, y + 5.5, x - 1.5, y + 7.7, x + 2.5, y + 9.5);
        heartShape.bezierCurveTo(x + 6, y + 7.7, x + 8, y + 5.5, x + 8, y + 3.5);
        heartShape.bezierCurveTo(x + 8, y + 3.5, x + 8, y, x + 5, y);
        heartShape.bezierCurveTo(x + 3.5, y, x + 2.5, y + 2.5, x + 2.5, y + 2.5);

        const extrudeSettings = { depth 1, bevelEnabled true, bevelSegments 2, steps 2, bevelSize 0.5, bevelThickness 0.5 };
        const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
        
        for (let i = 0; i  25; i++) {
            const material = new THREE.MeshStandardMaterial({
                color Math.random()  0.5  0xff0055  0xe8104a,
                metalness 0.5,
                roughness 0.3,
            });
            const heart = new THREE.Mesh(geometry, material);
            
            heart.position.set(
                (Math.random() - 0.5)  30,
                (Math.random() - 0.5)  20,
                (Math.random() - 0.5)  10
            );
            
            const scale = Math.random()  0.05 + 0.02;
            heart.scale.set(scale, scale, scale);
            
            heart.rotation.set(Math.random()  Math.PI, Math.random()  Math.PI, Math.random()  Math.PI);
            
            heart.userData = {
                baseY heart.position.y,
                driftX (Math.random() - 0.5)  0.01,
                driftY (Math.random()  0.01) + 0.005,
                rotX (Math.random() - 0.5)  0.01,
                rotY (Math.random() - 0.5)  0.01
            };
            
            hearts.push(heart);
            scene.add(heart);
        }
    }

    function animate() {
        requestAnimationFrame(animate);

        const time = Date.now()  0.0005;
        
        hearts.forEach(heart = {
            if (!finaleTriggered) {
                 heart.rotation.x += heart.userData.rotX;
                 heart.rotation.y += heart.userData.rotY;

                 heart.position.x += heart.userData.driftX;
                 heart.position.y = heart.userData.baseY + Math.sin(time + heart.position.x)  2;
                
                if (heart.position.y  12) heart.position.y = -12;
                if (heart.position.x  17) heart.position.x = -17;
                if (heart.position.x  -17) heart.position.x = 17;

            } else {
                  Final animation handled by GSAP
            }
        });

        renderer.render(scene, camera);
    }
    
    function triggerHeartFinale() {
        finaleTriggered = true;
        hearts.forEach(heart = {
            gsap.to(heart.position, {
                x heart.position.x + (Math.random() - 0.5)  10,
                y heart.position.y + 15 + Math.random()  5,
                z heart.position.z + (Math.random() - 0.5)  10,
                duration 4,
                ease power2.inOut
            });
            gsap.to(heart.rotation, {
                x Math.random()  10,
                y Math.random()  10,
                duration 4,
                ease power2.inOut
            });
            gsap.to(heart.material, {
                opacity 0,
                duration 2,
                delay 2,
                onComplete () = {
                   scene.remove(heart);
                }
            });
        });
    }

    window.addEventListener('resize', () = {
        camera.aspect = window.innerWidth  window.innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(window.innerWidth, window.innerHeight);
    });

     --- GSAP & STEP NAVIGATION ---
    let currentStep = 1;
    const totalSteps = 5;
    const progressBar = document.getElementById('progress-bar');
    const steps = document.querySelectorAll('.step');
    const nextButtons = document.querySelectorAll('.next-btn');
    
    function setupSteps() {
        gsap.set(steps, { autoAlpha 0, scale 0.8, display 'none' });
        gsap.set(steps[0], { autoAlpha 1, scale 1, display 'block' });
        updateProgressBar();
    }
    
    function goToStep(stepNumber) {
        if (stepNumber  totalSteps) return;

        const currentStepEl = document.querySelector(`.step[data-step=${currentStep}]`);
        const nextStepEl = document.querySelector(`.step[data-step=${stepNumber}]`);

        const tl = gsap.timeline({
            onComplete () = {
                currentStep = stepNumber;
                updateProgressBar();
            }
        });

        tl.to(currentStepEl, { autoAlpha 0, scale 0.8, duration 0.5, ease 'power2.in' })
          .set(currentStepEl, { display 'none' })
          .set(nextStepEl, { display 'block', autoAlpha 0, scale 0.8 })
          .to(nextStepEl, { autoAlpha 1, scale 1, duration 0.5, ease 'power2.out' })
          .from(nextStepEl.querySelectorAll('.step-content  '), {
              y 30,
              opacity 0,
              duration 0.5,
              stagger 0.1,
              ease 'power2.out'
          }, -=0.3);
    }

    function updateProgressBar() {
        const progress = ((currentStep - 1)  (totalSteps -1))  100;
        gsap.to(progressBar, { width `${progress}%`, duration 0.7, ease 'power2.out' });
    }

    nextButtons.forEach(button = {
        button.addEventListener('click', () = {
            goToStep(currentStep + 1);
        });
    });

     --- POLAROID TILT EFFECT ---
    const polaroidContainer = document.getElementById('polaroid-container');
    const polaroid = document.getElementById('polaroid');
    
    polaroidContainer.addEventListener('mousemove', (e) = {
        const rect = polaroidContainer.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        
        const centerX = rect.width  2;
        const centerY = rect.height  2;
        
        const rotateX = ((y - centerY)  centerY)  -10;
        const rotateY = ((x - centerX)  centerX)  10;
        
        gsap.to(polaroid, {
            rotationX rotateX,
            rotationY rotateY,
            duration 0.5,
            ease 'power1.out'
        });
    });
    
    polaroidContainer.addEventListener('mouseleave', () = {
        gsap.to(polaroid, {
            rotationX 0,
            rotationY 0,
            duration 0.5,
            ease 'elastic.out(1, 0.5)'
        });
    });

     --- FINALE CELEBRATION ---
    const celebrateBtn = document.getElementById('celebrate-btn');
    const finalMessage = document.getElementById('final-message');

    celebrateBtn.addEventListener('click', () = {
        gsap.to(celebrateBtn, { autoAlpha 0, scale 0.8, duration 0.3, onComplete () = celebrateBtn.disabled = true });
        
        gsap.to(finalMessage, { y -10, opacity 1, duration 0.5, delay 0.3, ease 'power2.out' });
        
        triggerConfetti();
        triggerHeartFinale();
    });

    function triggerConfetti() {
        const duration = 5  1000;
        const animationEnd = Date.now() + duration;
        const defaults = { startVelocity 30, spread 360, ticks 60, zIndex 100 };

        function randomInRange(min, max) {
            return Math.random()  (max - min) + min;
        }

         Powerful bursts from corners
        confetti({
            particleCount 150,
            spread 90,
            origin { x 0, y 1 },
            angle 45
        });
        confetti({
            particleCount 150,
            spread 90,
            origin { x 1, y 1 },
            angle 135
        });
        
         Continuous shower
        const interval = setInterval(function() {
            const timeLeft = animationEnd - Date.now();
            if (timeLeft = 0) {
                return clearInterval(interval);
            }

            const particleCount = 50  (timeLeft  duration);
            
             Standard confetti
            confetti(Object.assign({}, defaults, { particleCount, origin { x randomInRange(0.1, 0.9), y Math.random() - 0.2 } }));
            
             Emoji confetti
            confetti({
                particleCount 5,
                spread 30,
                origin { x randomInRange(0.1, 0.9), y Math.random() - 0.2 },
                shapes ['text'],
                shapeOptions {
                    text {
                        value ['‚ù§Ô∏è', 'üíñ', '‚ú®'],
                        size 40,
                        fontFamily 'sans-serif'
                    }
                }
            });
        }, 250);
    }

     Initialize everything
    document.addEventListener('DOMContentLoaded', () = {
        initThree();
        setupSteps();
    });
